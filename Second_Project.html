df_temp = df['Weekday']
## Now i will Write a function to either set 0 or 1 if week day or not respectively :
def isWeekday(x):
    weekend = ['Friday' , 'Thursday']
    if x in weekend:
        x = 0
    else:
        x = 1
    return x


df['Weekday'] = df_temp.apply(isWeekday)
